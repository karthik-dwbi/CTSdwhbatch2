1.SELECT d.department_name, COUNT(employee_id) AS no_of_employees
  FROM EMPLOYEES E JOIN departments d
  ON e.department_id=d.department_id
  GROUP BY DEPARTMENT_NAME
  HAVING COUNT(EMPLOYEE_ID)>=20;

2. SELECT EMPLOYEE_ID, FIRST_NAME, LAST_NAME, SALARY 
   FROM EMPLOYEES 
   WHERE SALARY > (SELECT AVG(SALARY) FROM EMPLOYEES E
                   WHERE D.DEPARTMENT_ID=E.DEPARTMENT_ID
                   GROUP BY E.EMPLOYEE_ID)

3. SELECT EMPLOYEE_ID, FIRST_NAME, LAST_NAME, CONCAT(FIRST_NAME, LAST_NAME) AS FULL_NAME, SALARY, (SALARY*0.1) AS NEW_SALARY, (SALARY-(SALARY*0.1)) AS DIFFERENCE
   FROM EMPLOYEES;

4. SELECT E.EMPLOYEE_ID, E.FIRST_NAME, E.LAST_NAME, B.FIRST_NAME AS MANAGER_NAME
   FROM EMPLOYEES E LEFT OUTER JOIN EMPLOYEES B
   ON E.MANAGER_ID = B.MANAGER_ID
   ORDER BY EMPLOYEE_ID;

5. SELECT EMPLOYEE_ID, FIRST_NAME, SALARY 
   FROM (SELECT * FROM EMPLOYEES
         ORDER BY SALARY DESC)
   WHERE ROWNUM<=3
   ORDER BY SALARY;
