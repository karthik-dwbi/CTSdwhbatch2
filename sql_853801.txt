1.
SELECT department_name,COUNT(EMPLOYEE_ID) AS NO_OF_EMPLOYEES FROM DEPARTMENTS JOIN EMPLOYEES
ON DEPARTMENTS.DEPARTMENT_ID=EMPLOYEES.DEPARTMENT_ID
GROUP BY DEPARTMENT_NAME
HAVING COUNT(EMPLOYEE_ID)>=20;


2.
SELECT employee_id,first_name,last_name,salary from employees a 
where salary > (select avg(salary) from employees b where a.department_id=b.department_id 
group by a.department_id);

3.
SELECT EMPLOYEE_ID,FIRST_NAME,LAST_NAME,(FIRST_NAME||' '||LAST_NAME) AS FULL_NAME,SALARY,(SALARY+(SALARY*0.1)) AS NEW_SALARY,
(SALARY+(SALARY*0.1) - (SALARY)) AS DIFFERENCE FROM employees;

4.
SELECT A.EMPLOYEE_ID,A.FIRST_NAME,A.LAST_NAME,B.FIRST_NAME AS MANAGER_NAME
FROM EMPLOYEES A 
LEFT OUTER JOIN EMPLOYEES B ON A.MANAGER_ID=B.MANAGER_ID
ORDER BY EMPLOYEE_ID;


5.
SELECT employee_id,first_name,salary FROM 
(SELECT * FROM EMPLOYEES)
WHERE ROWNUM<= 3
ORDER BY SALARY DESC;
