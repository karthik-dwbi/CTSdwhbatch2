SELECT DEPARTMENT_NAME,COUNT(EMPLOYEE_ID)
FROM DEPARTMENTS,EMPLOYEES
WHERE DEPARTMENTS.DEPARTMENT_ID=EMPLOYEES.DEPARTMENT_ID
GROUP BY DEPARTMENT_NAME
HAVING COUNT(EMPLOYEE_ID)>=20;



SELECT EMPLOYEE_ID,FIRST_NAME,LAST_NAME,SALARY
FROM EMPLOYEES A
WHERE SALARY >(SELECT AVG(SALARY)
FROM EMPLOYEES B
WHERE A.DEPARTMENT_ID = B.DEPARTMENT_ID
GROUP BY A.DEPARTMENT_ID)


SELECT EMPLOYEE_ID,FIRST_NAME,LAST_NAME,
(FIRST_NAME||''||LAST_NAME)
FULL_NAME,SALARY,(SALARY*0.1)NEW_SALARY,
(SALARY-(SALARY*0.1))
DIFFERENCE FROM EMPLOYEES;




SELECT A.EMPLOYEE_ID,A.FIRST_NAME,A.LAST_NAME,B.FIRST_NAME AS MANAGER_NAME
FROM EMPLOYEES A LEFT OUTER JOIN EMPLOYEES B
ON A.MANAGER_ID=B.MANAGER_ID
ORDER BY EMPLOYEE_ID;




SELECT EMPLOYEE_ID,FIRST_NAME,SALARY
FROM (SELECT*
      FROM EMPLOYEES
      ORDER BY SALARY DESC)
WHERE ROWNUM<=3
ORDER BY SALARY;
